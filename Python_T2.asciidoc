    if len(text) < 8:
        return False
    has_uppercase = False
    has_lowercase = False

        elif ch.islower():
            has_lowercase = True
        elif ch.isdigit():
            has_digit = True

def fun(x):

    length = len(x)
    even_sum = 0
    odd_sum = 0
    for i in range(length):
        if i%2==0:
            even_sum += x[i]
        else:
            odd_sum += x[i]
    count = 0 
    left_odd = 0
    left_even = 0 
    for i in range(length):
        if i%2==0:
            even_sum -= x[i]
        else:
            odd_sum -= x[i]
        if odd_sum + left_even == even_sum + left_odd:
            count+=1
        if i%2==0:
            left_even += x[i]
        else:
            left_odd += x[i]
    return count

def valid(s):
    stack = []
    pairs = {')':'(',']':'[','}':'{'}
    for ch in s:
        if ch in "({[":
            stack.append(ch)
        elif ch in ")}]":
            if not stack or stack[-1] != pairs[ch]:
                return False 
            stack.pop()
    return len(stack)==0
